61  __host__ __device__ GReal_t pdk(const GReal_t a, const GReal_t b,
62  const GReal_t c) const
63  {
64  //the PDK function
65  GReal_t x = (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c);
66  x = sqrt(x) / (2 * a);
67  return x;
68  }

70  __host__ __device__ void bbsort(GReal_t *array, GInt_t n)
71  {
72  // Improved bubble sort
73  for (GInt_t c = 0; c < n; c++)
74  {
75  GInt_t nswap = 0;
76 
77  for (GInt_t d = 0; d < n - c - 1; d++)
78  {
79  if (array[d] > array[d + 1]) /* For decreasing order use < */
80  {
81  GReal_t swap = array[d];
82  array[d] = array[d + 1];
83  array[d + 1] = swap;
84  nswap++;
85  }
86  }
87  if (nswap == 0)
88  break;
89  }

93  __host__ __device__ GReal_t process(const GInt_t evt, Vector4R** particles)
94  {
95 
96  GReal_t fTeCmTm = 0.0, fWtMax = 0.0;
97 
98  fTeCmTm = particles[0]->mass(); // total energy in C.M. minus the sum of the masses
99 
100 #pragma unroll 9
101  for (size_t n = 0; n < fNDaughters; n++)
102  {
103  fTeCmTm -= fMasses[n];
104  }
105 
106  GReal_t emmax = fTeCmTm + fMasses[0];
107  GReal_t emmin = 0.0;
108  GReal_t wtmax = 1.0;
109 
110 #pragma unroll 9
111  for (size_t n = 1; n < fNDaughters; n++)
112  {
113  emmin += fMasses[n - 1];
114  emmax += fMasses[n];
115  wtmax *= pdk(emmax, emmin, fMasses[n]);
116  }
117  fWtMax = 1.0 / wtmax;
118  //
119  //----> get the betas of the decaying particle
120  //
121  GReal_t fBeta[3];
122  fBeta[0]=0, fBeta[1]=0, fBeta[2] = 0.0;
123 
124  GReal_t _beta = particles[0]->d3mag() / particles[0]->get(0);
125 
126  if (_beta)
127  {
128  GReal_t w = _beta / particles[0]->d3mag();
129  fBeta[0] = particles[0]->get(0) * w;
130  fBeta[1] = particles[0]->get(1) * w;
131  fBeta[2] = particles[0]->get(2) * w;
132  }
133 
134  GReal_t rno[kMAXP];
135  rno[0] = 0.0;
136 
137  if (fNDaughters > 2)
138  {
139 #pragma unroll 9
140  for (size_t n = 1; n < fNDaughters - 1; n++)
141  rno[n] = fRandNumbers[n - 1 + evt * fNDaughters];
142  bbsort(&rno[1], fNDaughters - 2);
143 
144  }
145  rno[fNDaughters - 1] = 1;
146 
147  GReal_t invMas[kMAXP], sum = 0.0;
148 
149 #pragma unroll 9
150  for (size_t n = 0; n < fNDaughters; n++)
151  {
152  sum += fMasses[n];
153  invMas[n] = rno[n] * fTeCmTm + sum;
154  }
155 
156  //
157  //-----> compute the weight of the current event
158  //
159 
160  GReal_t wt = 1.0 / wtmax;
161 
162  GReal_t pd[kMAXP];
163 
164 #pragma unroll 9
165  for (size_t n = 0; n < fNDaughters - 1; n++)
166  {
167  pd[n] = pdk(invMas[n + 1], invMas[n], fMasses[n + 1]);
168  wt *= pd[n];
169  }
170 
171  //
172  //-----> complete specification of event (Raubold-Lynch method)
173  //
174 
175  particles[1]->set(sqrt(pd[0] * pd[0] + fMasses[0] * fMasses[0]), 0.0,
176  pd[0], 0.0);
177 
178 #pragma unroll 9
179  for (size_t i = 1; i < fNDaughters; i++)
180  {
181 
182  particles[i + 1]->set(
183  sqrt(pd[i - 1] * pd[i - 1] + fMasses[i] * fMasses[i]), 0.0,
184  -pd[i - 1], 0.0);
185 
186  GReal_t cZ = 2
187  * fRandNumbers[i - 1 + fNDaughters - 1 + evt * fNDaughters]
188  - 1;
189  GReal_t sZ = sqrt(1 - cZ * cZ);
190  GReal_t angY = 2.0 * PI
191  * fRandNumbers[i + fNDaughters - 1 + evt * fNDaughters];
192  GReal_t cY = cos(angY);
193  GReal_t sY = sin(angY);
194  for (size_t j = 0; j <= i; j++)
195  {
196 
197  GReal_t x = particles[j + 1]->get(1);
198  GReal_t y = particles[j + 1]->get(2);
199  particles[j + 1]->set(1, cZ * x - sZ * y);
200  particles[j + 1]->set(2, sZ * x + cZ * y); // rotation around Z
201 
202  x = particles[j + 1]->get(1);
203  GReal_t z = particles[j + 1]->get(3);
204  particles[j + 1]->set(1, cY * x - sY * z);
205  particles[j + 1]->set(3, sY * x + cY * z); // rotation around Y
206  }
207 
208  if (i == (fNDaughters - 1))
209  break;
210 
211  GReal_t beta = pd[i] / sqrt(pd[i] * pd[i] + invMas[i] * invMas[i]);
212  for (size_t j = 0; j <= i; j++)
213  {
214 
215  particles[j + 1]->applyBoostTo(0, beta, 0);
216  }
217 
218  //i++;
219  }
220 
221  //
222  //---> final boost of all particles to the mother's frame
223  //
224 #pragma unroll 9
225  for (size_t n = 0; n < fNDaughters; n++)
226  {
227 
228  particles[n + 1]->applyBoostTo(*particles[0]);
229 
230  //printf("[%d] %f\n", n, particles[n+1]->mass() );
231  }
232 
233  //
234  //---> return the weight of event
235  //
236 
237  return wt;
238 
239  }
